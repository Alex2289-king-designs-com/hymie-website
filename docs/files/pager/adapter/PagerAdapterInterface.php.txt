namespace hymie\pager\adapter;

/**
 * 分页适配器接口
 */
interface PagerAdapterInterface
{
    /**
     * 获取数据总条数
     *
     * @return void
     */
    public function getDataCount();

    /**
     * 按照分页获取数据列表
     *
     * @param int $page
     * @param int $pageSize
     * @return array
     */
    public function getData($page, $pageSize);
}
